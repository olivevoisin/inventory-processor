/**

 * @param {string} userId - ID de l'utilisateur
 * @param {string} sessionId - ID de session optionnel
 * @returns {Promise<Object>} - Résultat de la reconnaissance

async function processAudioSegment(filePath, location, userId, sessionId) {
  try {
    logger.info(`Traitement du segment audio: ${filePath} pour l'emplacement: ${location}`);

/**
 * Transcrire un fichier audio en texte
 * @param {Buffer} audioData - Données audio
 * @returns {Promise<Object>} - Résultat de la transcription
 */
async function transcribeAudio(audioData) {
  try {
    const mimetype = 'audio/wav'; // Dans une implémentation réelle, cela serait détecté dynamiquement
    
    // Envoyer l'audio à Deepgram
    const response = await deepgram.transcription.preRecorded({
      buffer: audioData,
      mimetype
    }, {
      punctuate: true,
      language: 'fr',
      model: 'nova-2'
    }).transcribe();
    
    // Extraire la transcription et la confiance
    const transcript = response.results.channels[0].alternatives[0].transcript;
    const confidence = response.results.channels[0].alternatives[0].confidence;
    
    return { transcript, confidence };
  } catch (error) {
    logger.error(`Erreur de transcription: ${error.message}`);
    throw new Error(`Échec de la transcription: ${error.message}`);
  }
}
 
/**
 * Extraire des éléments d'inventaire d'une transcription
 * @param {string} transcript - Texte transcrit
 * @returns {Array<Object>} - Éléments d'inventaire extraits
 */
function extractInventoryItems(transcript) {
  if (!transcript) return [];
  
  const items = [];
  const text = transcript.toLowerCase();
  
  // Regex pour les patterns comme "X [unité] de [produit]"
  // Ex: "cinq bouteilles de vin rouge", "3 cannettes de bière"
  const bottlePattern = /(\w+|[0-9]+)\s+(bouteilles?)\s+(?:de\s+)?([a-zéèêëàâäôöùûüïîçñ\s]+)/gi;
  const canPattern = /(\w+|[0-9]+)\s+(cannettes?)\s+(?:de\s+)?([a-zéèêëàâäôöùûüïîçñ\s]+)/gi;
  const boxPattern = /(\w+|[0-9]+)\s+(boîtes?|caisses?)\s+(?:de\s+)?([a-zéèêëàâäôöùûüïîçñ\s]+)/gi;
  
  // Fonction pour extraire les items selon un pattern
  const extractItemsWithPattern = (pattern, unitType) => {
    let match;
    while ((match = pattern.exec(text)) !== null) {
      // Extraire la quantité
      let quantity;
      if (isNaN(match[1])) {
        quantity = wordToNumber[match[1]] || 1;
      } else E{
        quantity = parseInt(match[1], 10);
      }
      
      // Extraire le nom du produit
      let productName = match[3].trim();
      
      // Première lettre en majuscule
      productName = productName.charAt(0).toUpperCase() + productName.slice(1);
      
      items.push({
        name: productName,
        quantity,
        unit: unitType
      });
    }
  };
  
  // Extraire tous les types d'items
  extractItemsWithPattern(bottlePattern, 'bouteille');
  extractItemsWithPattern(canPattern, 'cannette');
  extractItemsWithPattern(boxPattern, 'boîte');
  
  return items;
}
 
/**
 * Détecter une commande vocale dans la transcription
 * @param {string} transcript - Texte transcrit
 * @returns {Object|null} - Commande détectée ou null
 */
function detectCommand(transcript) {
  if (!transcript) return null;
  
  const text = transcript.toLowerCase();
  
  // Commandes de terminaison
  if (text.includes('terminer') || text.includes('fin') || text.includes('fini')) {
    return { type: 'end', text: 'Terminer l\'inventaire' };
  }
  
  // Commandes d'annulation
  if (text.includes('annuler') || text.includes('annule')) {
    return { type: 'cancel', text: 'Annuler la dernière saisie' };
  }
  
  // Commandes de confirmation
  if (text.includes('confirmer') || text.includes('valider') || text.includes('oui')) {
    return { type: 'confirm', text: 'Confirmer la saisie' };
  }
  
  // Commandes de rejet
  if (text.includes('refuser') || text.includes('non')) {
    return { type: 'reject', text: 'Rejeter la proposition' };
  }
  
  // Commandes d'aide
  if (text.includes('aide') || text.includes('help')) {
    return { type: 'help', text: 'Afficher l\'aide' };
  }
  
  return null;
}

/**
 * Calcule un score de calibration
 * @param {string} transcript - Texte transcrit
 * @param {Object} testPhrases - Phrases de test attendues
 * @returns {Object} - Score et recommandations
 */
function calculateCalibrationScore(transcript, testPhrases) {
  // Dans une implémentation réelle, nous comparerions le transcript avec les phrases de test
  const score = Math.random() * 0.3 + 0.7; // Score aléatoire entre 0.7 et 1.0 pour les tests
  
  // Générer des recommandations
  let recommendations = [];
  if (score > 0.9) {
    recommendations.push("Excellente reconnaissance!");
  } else if (score > 0.8) {
    recommendations.push("Bonne reconnaissance. Essayez de parler un peu plus clairement.");
  } else {
    recommendations.push("Reconnaissance acceptable. Parlez plus distinctement et réduisez les bruits de fond.");
  }
  
  return {
    overallScore: score,
    recommendations
  };
}
